import { DB, UserModelDB } from '@app/server/Service/DatabaseService';
import { P } from '@app/type/framework/data/P';
 
 
export const selectUserByClerkUserId = async (clerkUserId : string) : P<Partial<UserModelDB> | null> =>
    DB.user.findFirst({
        where : { clerkUserId },
    });
 
export const updateUserClerkUserId = async (
    user : UserModelDB,
    clerkUserId : string,
) : P<Partial<UserModelDB>> =>
    DB.user.update({
        where : { id : user.id },
        data  : { clerkUserId },
    });
