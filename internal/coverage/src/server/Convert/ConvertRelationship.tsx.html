import { RelationshipType } from '@app/constant/enum/model/RelationshipTypeEnum';
import { DefaultModelStash } from '@app/constant/default/DefaultModelStash';
import { RelationshipModelDB } from '@app/server/Service/DatabaseService';
import { RelationshipModel } from '@app/type/model/RelationshipModel';
import { ModelType } from '@app/constant/enum/model/ModelTypeEnum';
import { ModelConverter } from '@app/type/model/ModelConverter';
import { ModelIdLookup } from '@app/type/model/ModelIdLookup';
import { CoreId } from '@app/type/framework/core/CoreId';
import { ModelStash } from '@app/type/model/ModelStash';
import {
    toDatabaseString, toModelSubtype, toDatabaseHash,
    toModelString,    toDatabaseId,   toModelDate,
    toModelHash,      toModelId,      toHashId,
} from './ConvertUtil';
 
 
const toListModel = <T,>(
    list         : Partial<RelationshipModelDB>[] | undefined,
    defaultValue : T = null as unknown as T,
) : RelationshipModel[] =>
    list
        ? list.map(toModel)
        : defaultValue as unknown as RelationshipModel[];
 
const toListDatabase = (
    list : RelationshipModel[],
) : Partial<RelationshipModelDB>[] =>
    list.map(toDatabase);
 
const toStashFromList = (
    modelList : RelationshipModel[],
    lookup : ModelIdLookup | null = null,
) : ModelStash =>
    modelList.reduce((stash, model) =>({
        ...stash,
        ...toStash(model, lookup),
    }), DefaultModelStash);
 
const toModel = (modelDB : Partial<RelationshipModelDB> = {}) : RelationshipModel => ({
 
    modelType : ModelType.Relationship,
 
    description : toModelString( modelDB.description, '') as string,
    createdAt   : toModelDate(   modelDB.createdAt      ),
    updatedAt   : toModelDate(   modelDB.updatedAt      ),
    createdBy   : toModelId(     modelDB.created_by      ),
    debug       : toModelString( modelDB.debug,       '') as string,
    title       : toModelString( modelDB.title,       '') as string,
    hash        : toModelHash(   modelDB.hash           ),
    id          : toModelId(     modelDB.id             ),
 
    relationshipType : toModelSubtype(modelDB.relationshipType, RelationshipType),
 
    sourceModelType : toModelSubtype(modelDB.sourceModelType, ModelType),
    sourceModelHash : toModelHash(   modelDB.sourceModelHash),
    sourceModelId   : toModelId(     modelDB.sourceModelId  ),
 
    destModelType : toModelSubtype(modelDB.destModelType, ModelType),
    destModelHash : toModelHash(   modelDB.destModelHash),
    destModelId   : toModelId(     modelDB.destModelId  ),
 
    contextDescription : toModelString( modelDB.contextDescription, '') as string,
 
});
 
const toDatabase = (model : Partial<RelationshipModel> = {}) : Partial<RelationshipModelDB> => ({
 
    description : toDatabaseString(model.description),
    created_by  : toDatabaseId(    model.createdBy  ),
    title       : toDatabaseString(model.title      ),
    debug       : toDatabaseString(model.debug      ),
    id          : toDatabaseId(    model.id         ),
    hash        : toDatabaseHash(  model.hash       ),
 
    relationshipType : toModelSubtype(model.relationshipType, RelationshipType) ,
 
    sourceModelId   : toDatabaseId(  model.sourceModelId),
    sourceModelHash : toDatabaseHash(model.sourceModelHash),
    sourceModelType : toModelSubtype(model.sourceModelType, ModelType),
 
    destModelId   : toDatabaseId(  model.destModelId),
    destModelHash : toDatabaseHash(model.destModelHash),
    destModelType : toModelSubtype(model.destModelType, ModelType),
 
    contextDescription : toDatabaseString(model.contextDescription),
});
 
const toStash = (
    model  : RelationshipModel,
    lookup : ModelIdLookup | null = null,
) : ModelStash => {
 
    lookup = lookup || {};
 
    lookup[`${model.modelType}_${model.id}`] = model.hash || '';
 
    const stash : ModelStash = DefaultModelStash;
 
    const relationship = toHashId(model);
 
    relationship.sourceModelId = relationship.sourceModelHash as CoreId;
    relationship.destModelId   = relationship.destModelHash as CoreId;
 
    relationship.createdBy = lookup[`${ModelType.User}_${relationship.createdBy}`] as unknown as CoreId
 
    stash.Relationship[relationship.hash as string] = relationship;
 
    return stash;
};
 
 
export const ConvertRelationship : ModelConverter<RelationshipModel, RelationshipModelDB> = {
    toStashFromList, toListDatabase, toListModel,
    toDatabase,      toStash,        toModel,
};
